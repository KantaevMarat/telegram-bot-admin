# Команды для исправления git конфликта

# Вариант 1: Сбросить локальные изменения (РЕКОМЕНДУЕТСЯ)
git checkout -- setup-ssl.sh
git pull origin sync/cleanup/2025-10-29
./setup-ssl.sh

# Вариант 2: Сохранить в stash
git stash
git pull origin sync/cleanup/2025-10-29
./setup-ssl.sh

# Вариант 3: Жесткий сброс (если ничего не помогло)
git reset --hard origin/sync/cleanup/2025-10-29
./setup-ssl.sh
