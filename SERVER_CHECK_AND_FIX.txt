# Команды для выполнения на сервере (по порядку)

# 1. Проверить что media.service.ts есть
ls -la backend/src/modules/media/

# 2. Если нет, получить последние изменения
git fetch origin
git pull origin sync/cleanup/2025-10-29

# 3. Проверить снова
ls -la backend/src/modules/media/

# 4. Пересобрать backend с выводом логов
docker compose build --no-cache backend 2>&1 | tee build.log

# 5. Если сборка успешна, запустить
docker compose up -d backend

# 6. Проверить логи
docker compose logs -f backend
